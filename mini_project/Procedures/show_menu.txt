CREATE OR REPLACE PROCEDURE SHOW_MENU(V_RES_ID RESTAURANT.RES_ID%TYPE) IS
	CURSOR CUR_MENU IS SELECT * FROM MENU WHERE RES_ID=V_RES_ID;
FLAG RESTAURANT.RES_ID%TYPE;
BEGIN
	SELECT RES_ID INTO FLAG FROM RESTAURANT WHERE RES_ID=V_RES_ID;
	FOR I IN CUR_MENU LOOP
		DBMS_OUTPUT.PUT_LINE(V_RES_ID||'----'||RPAD(I.ITEM_NAME,40)||' RS.'||I.ITEM_COST);
	END LOOP;
	DBMS_OUTPUT.PUT_LINE('TO PLACE AN ORDER, WRITE');
	DBMS_OUTPUT.PUT_LINE('EXEC PLACE_ORDER(<YOUR MOBILE NUMBER>,<RES_ID>,<ITEM_NAME>,<QTY>)');
EXCEPTION
	WHEN NO_DATA_FOUND THEN
		DBMS_OUTPUT.PUT_LINE('WRONG RESTAURANT ID');
END;
